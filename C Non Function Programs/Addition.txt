Q1:addition of 2 numbers
-------------------------
		#include <stdio.h>
		int main()
		{
			int number1,number2,result;
			number1=10;
			number2=20;
			result=number1+number2;
			printf("%d\n",result);
			number1=100;
			number2=200;
			result=number1+number2;
			printf("%d\n",result);
			number1=1000;
			number2=2000;
			result=number1+number2;
			printf("%d\n",result);
		}
