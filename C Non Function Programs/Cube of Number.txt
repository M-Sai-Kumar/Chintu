Q4:print cube of a number
-------------------------
		#include <stdio.h>
		int main()
		{
		int number,cube;
			number=5;
			cube=number*number*number;
			printf("%d\n",cube);
			number=3;
			cube=number*number*number;
			printf("%d\n",cube);
			number=2;
			cube=number*number*number;
			printf("%d\n",cube);
		}
